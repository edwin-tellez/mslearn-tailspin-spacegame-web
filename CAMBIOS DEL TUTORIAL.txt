dotnet test --no-build  --configuration Release /p:CollectCoverage=true  /p:CoverletOutputFormat=cobertura /p:CoverletOutput=./TestResults/Coverage/

Para dotnet core 3.x
 dotnet tool install --global dotnet-reportgenerator-globaltool --version 4.4.7

Para convertir los reportes:
reportgenerator.exe -reports:./Tailspin.SpaceGame.Web.Tests/TestResults/Coverage/coverage.cobertura.xml -targetdir:./CodeCoverage -reporttypes:HtmlInline_AzurePipelines

En lugar de:
 $HOME/.dotnet/tools/reportgenerator \
  -reports:./Tailspin.SpaceGame.Web.Tests/TestResults/Coverage/coverage.cobertura.xml \
  -targetdir:./CodeCoverage \
  -reporttypes:HtmlInline_AzurePipelines
